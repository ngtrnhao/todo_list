Quy tắc đặt tên đầy đủ trong React/JavaScript
1. Quy tắc chung
1.1. Kiểu chữ (Case Types)
// PascalCase: Cho Components/Classes
class UserProfile {}
function TodoList {}

// camelCase: Cho biến, functions, props
const firstName = 'John';
const handleSubmit = () => {};

// SNAKE_CASE: Cho constants
const MAX_COUNT = 100;
const API_KEY = 'xyz';

// kebab-case: Cho CSS classes
.user-profile {}
.nav-item {}
